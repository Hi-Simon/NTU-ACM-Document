	\item subtract(); 相减；
	\item multiply(); 相乘；
	\item divide(); 相除取整；
	\item remainder(); 取余；
	\item pow();   a.pow(b)=a\^b；
	\item gcd();   最大公约数；
	\item abs(); 绝对值；
	\item negate(); 取反数；
	\item mod(); a.mod(b)=a\%b=a.remainder(b)；

	\item public int comareTo()；
	\item boolean equals(); 是否相等；
	\item BigInteger(String val); 将指定字符串转换为十进制表示形式；
